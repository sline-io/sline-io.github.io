Exemples d'utilisation du SDK Sline

  • Location mono-produit FNAC
  • Location import de panier FNAC
  • Location mono-produit Simon Simone
  • Location import de panier Simon Simone 
              
  // À mettre dans la section <head> de la page
  <script type="text/javascript" src="https://cdn.jsdelivr.net/gh/sline-io/web-sdk@1/sline-sdk.min.js"></script>
  
  // À mettre dans la section <body> de la page
  <script type="text/javascript">

    /* Initialisation du SDK
     * Initialize(string slug, boolean isProduction)
     */
    Sline.Initialize('simonsimone', false);
    
    /* Configuration du panier à importer dans le parcours locatif
     * AddCart(string sku, int quantity)
     */
    Sline.AddCart('784-3328-3289-3296-3325-3327', 1);
    Sline.AddCart('13103', 4);
    
    /*
     * Déclenche un call API, puis injecte l'url du parcours de souscription dans le lien avec l'ID spécifié en paramètre
     * et le texte du second paramètre (optionnel) en prefixe de "X €/mois" dans le texte du lien
     */
     Sline.RequestCheckoutURL("rent-btn", "Louer à partir de ");

    /*
     * Optionnel - afin de vider complètement le panier
     */
    Sline.ResetCart();

    /* 
     * Mise à jour du panier et du lien vers le parcours locatif à insérer dans le callback déclenché au click sur les sélecteurs de quantité
     */
    Sline.UpdateCart(sku, newQuantity);
    Sline.RequestCheckoutURL("rent-btn", "Louer à partir de ");
  </script>

  // La balise où injecter l'url du parcours de souscription
  <a href="#" id="rent-btn">Louer</a>
